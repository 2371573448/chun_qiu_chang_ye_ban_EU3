#中山专属buff
country_event = { 
	id = 9300
	trigger = { 
		tag = NPC
		any_country = { 
			tag = ZHS
			any_owned_province = { 
				not = { 
					has_province_modifier = chengzhongyoushan1
				}
			}
		}
	}
	mean_time_to_happen = { 
		months = 1
	}
	title = "EVTNAME9300"
	desc = "EVTDESC9300"
	option = { 
		name = "EVTOPTA9300"
		any_country = { 
			limit = { 
				tag = ZHS
			}
			any_owned = { 
				limit = { 
					has_building = fort4
					not = { 
						has_province_modifier = chengzhongyoushan1
					}
				}
				add_province_modifier = { 
					name = "chengzhongyoushan1"
					duration = -1
				}
			}
		}
	}
}
