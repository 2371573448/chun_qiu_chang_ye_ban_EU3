			OR = {

					trade_goods = spices

				not = { 
					has_province_modifier = tribe_product1
				}
			}
		}
	}
	mean_time_to_happen = { 
		months = 1
	}
	title = "EVTNAME1551"
	desc = "EVTDESC1551"

		name = "EVTOPTA1551"

			limit = { 
				OR = {
